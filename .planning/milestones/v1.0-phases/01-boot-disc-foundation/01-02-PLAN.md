```javascript
import { Howl } from 'howler'

let sound = null

export function initAudio() {
  if (sound) return sound
  sound = new Howl({
    src: ['/audio/background.mp3'],
    loop: true,
    volume: 0,
    autoplay: false,
  })
  return sound
}

export function playWithFade(duration = 2000) {
  const s = initAudio()
  s.play()
  s.fade(0, 1, duration)
}

export function setMuted(muted) {
  if (sound) sound.mute(muted)
}

export function getSound() {
  return sound
}
```

Key points:
- Module-level `sound` variable ensures singleton (never create multiple Howl instances)
- `volume: 0` start — `playWithFade` brings it up over 2 seconds

- Audio file path: `/audio/background.mp3` (served from `public/audio/`)
- Howler's `onplayerror` handling will be in BootSequence.jsx (the consumer)

Component structure:
- Full-viewport absolutely positioned `<div>` with `zIndex: 20`, black background
- Inside: a `<canvas>` element filling the viewport (ref-based, not R3F Canvas)
- Uses `requestAnimationFrame` loop in a `useEffect` for animation
- On completion: calls `setPhase(2)` via Zustand and sets sessionStorage flag

**Animation timeline (elapsed time from mount):**

1. **Letter Tracing Phase (0ms - 2000ms):**
   - Canvas renders "LOADING" text centered on viewport
   - Use canvas `measureText` to get total text width, center it
   - Two "sprites" (bright glowing dots) sweep horizontally across the text
   - Cyan sprite reveals the top half of each letter (use `ctx.save()`, `ctx.beginPath()`, `ctx.rect(0, 0, canvas.width, centerY)`, `ctx.clip()`)
   - Orange sprite reveals the bottom half (clip rect from centerY to canvas.height)
   - Progress variable `t = elapsed / 2000` drives the horizontal reveal: a vertical clip line sweeps from left edge of text to right edge
   - Leading glow dot: draw a bright circle (radial gradient, white center fading to cyan/orange) at the sweep position
   - Font: load TR2N font via `ctx.font = '${fontSize}px TR2N'` where fontSize is `Math.min(canvas.width * 0.12, 120)`
   - Stroke style: cyan `#00FFFF` for top half, orange `#FF5E00` for bottom half
   - Use `ctx.strokeText` with `lineWidth: 2` for the neon outline look
   - Also use `ctx.shadowBlur = 15` and `ctx.shadowColor` matching stroke color for glow

2. **Collision Flash Phase (2000ms - 2200ms):**
   - Both sprites converge to center of text
   - White radial gradient flash expands from center to fill viewport
   - Use `easeOutQuad(t) = 1 - (1-t)*(1-t)` for expansion timing
   - `maxRadius = Math.sqrt((canvas.width/2)**2 + (canvas.height/2)**2) * 1.5`
   - Flash gradient: white center, fading to transparent at edges

3. **Fade to Black (2200ms - 2700ms):**
   - Canvas fills with black at increasing opacity
   - At 2500ms: call `playWithFade(2000)` from audioManager to start music
   - Handle autoplay failure: listen for Howler `playerror` event, show "CLICK TO ENABLE AUDIO" text on canvas; on any document click, retry `playWithFade()`

4. **Title Reveal Phase (2700ms - 4200ms):**
   - Black canvas background
   - "TANMAY GOEL" fades in at center — font: TR2N, color: `#FF0000`, text-shadow glow
   - "SOFTWARE DEVELOPER" fades in below — font: Roboto Mono, color: `#F0F0F0`
   - Both use `ctx.globalAlpha` increasing from 0 to 1 over 1000ms
   - Text shadow: `ctx.shadowBlur = 20`, `ctx.shadowColor = 'rgba(255,0,0,0.6)'` for title

5. **Transition Phase (4200ms - 4700ms):**
   - Title fades out (globalAlpha decreasing)
   - At 4700ms: call `onComplete()` callback

**Update `src/App.jsx`** — add BootSequence overlay, phase-conditional rendering, and audio mute subscription.

Replace the entire App component with:

```jsx
import { useState, useEffect, useMemo } from 'react'
import { Canvas } from '@react-three/fiber'
import Scene from './components/Scene'
import TitleOverlay from './components/UI/TitleOverlay'
import BootSequence from './components/UI/BootSequence'
import useAppState from './store/appState'
import { setMuted } from './utils/audioManager'

function App() {
  const [discHovered, setDiscHovered] = useState(false)
  const phase = useAppState((s) => s.phase)

  // Subscribe to audioEnabled changes — sync with Howler mute
  useEffect(() => {
    const unsub = useAppState.subscribe(
      (s) => s.audioEnabled,
      (enabled) => setMuted(!enabled)
    )
    return unsub
  }, [])

  return (
    <div style={{ width: '100vw', height: '100vh', background: '#000' }}>
      {phase === 1 && <BootSequence />}
      <Canvas
        camera={{ position: [0, 0, 8], fov: 60 }}
        gl={{ antialias: true, alpha: false }}
        dpr={[1, 2]}
        onPointerMissed={() => setDiscHovered(false)}
        style={{ position: 'absolute', top: 0, left: 0 }}
      >
        <Scene onDiscHover={setDiscHovered} />
      </Canvas>
      {phase >= 2 && <TitleOverlay glitch={discHovered} visible={phase === 2} />}
    </div>
  )
}

export default App
```
