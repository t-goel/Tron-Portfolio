Create `src/components/UI/EnterButton.jsx`.

The button is a Tron-styled CTA positioned in the lower-center of the viewport, below the disc. It uses CSS-in-JS inline styles and a hover state managed via React useState.

```jsx
import { useState } from 'react'
import useAppState from '../../store/appState'

export default function EnterButton() {
  const setPhase = useAppState((s) => s.setPhase)
  const [hovered, setHovered] = useState(false)

  return (
    <div
      style={{
        position: 'absolute',
        bottom: '12%',
        left: '50%',
        transform: 'translateX(-50%)',
        zIndex: 10,
        pointerEvents: 'auto',
      }}
    >
      <button
        onClick={() => setPhase(3)}
        onMouseEnter={() => setHovered(true)}
        onMouseLeave={() => setHovered(false)}
        style={{
          background: 'transparent',
          border: `1px solid ${hovered ? '#FF0000' : 'rgba(255, 0, 0, 0.35)'}`,
          color: hovered ? '#FF0000' : 'rgba(255, 0, 0, 0.55)',
          fontFamily: "'TR2N', sans-serif",
          fontSize: 'clamp(0.65rem, 1.4vw, 0.9rem)',
          letterSpacing: '0.35em',
          padding: '0.75rem 2.5rem',
          cursor: 'pointer',
          userSelect: 'none',
          whiteSpace: 'nowrap',
          textShadow: hovered
            ? '0 0 12px rgba(255, 0, 0, 0.9), 0 0 24px rgba(255, 0, 0, 0.5)'
            : 'none',
          boxShadow: hovered
            ? '0 0 12px rgba(255, 0, 0, 0.4), inset 0 0 12px rgba(255, 0, 0, 0.07)'
            : 'none',
          transform: hovered ? 'scale(1.04)' : 'scale(1)',
          transition: 'all 0.2s ease',
          outline: 'none',
        }}
      >
        ENTER THE GRID
      </button>
    </div>
  )
}
```

Key details:
- `bottom: '12%'` places it below the disc (which is centered vertically)
- Transparent background — blends with the 3D scene behind it
- Border and text start dim (`rgba(255, 0, 0, 0.35)`) and brighten to full `#FF0000` on hover
- `textShadow` + `boxShadow` on hover create the neon glow
- `scale(1.04)` on hover gives a subtle physical response
- `transition: 'all 0.2s ease'` smoothly animates all changes
- `pointerEvents: 'auto'` on the wrapper (parent TitleOverlay has `pointerEvents: 'none'` — this is a separate component so no conflict)
