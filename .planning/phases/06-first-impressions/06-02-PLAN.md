Create new file `src/components/UI/GridAffordanceHint.jsx` with the following implementation:

```javascript
import { useState, useEffect } from 'react'

const SESSION_KEY = 'grid_hint_shown'

export default function GridAffordanceHint() {
  const [visible, setVisible] = useState(false)
  const [fadingOut, setFadingOut] = useState(false)

  useEffect(() => {
    if (sessionStorage.getItem(SESSION_KEY)) return
    sessionStorage.setItem(SESSION_KEY, '1')
    setVisible(true)

    const fadeTimer = setTimeout(() => setFadingOut(true), 3000)
    const hideTimer = setTimeout(() => setVisible(false), 3600)
    return () => {
      clearTimeout(fadeTimer)
      clearTimeout(hideTimer)
    }
  }, [])

  if (!visible) return null

  return (
    <div
      style={{
        position: 'fixed',
        bottom: '80px',
        left: '50%',
        transform: 'translateX(-50%)',
        zIndex: 25,
        opacity: fadingOut ? 0 : 1,
        transition: 'opacity 0.6s ease',
        pointerEvents: 'none',
        fontFamily: "'Roboto Mono', monospace",
        color: '#00FFFF',
        fontSize: '0.75rem',
        letterSpacing: '0.2em',
        textTransform: 'uppercase',
        textShadow: '0 0 8px #00FFFF',
      }}
    >
      Click and drag to explore
    </div>
  )
}
```

**Key details:**
- sessionStorage key: `'grid_hint_shown'` — checked on mount, set to `'1'` before showing
- Fade starts at T+3000ms (fadingOut = true), component removes at T+3600ms (visible = false)
- CSS transition: `opacity 0.6s ease` handles the fade animation
- `pointerEvents: 'none'` so it doesn't block 3D canvas interaction
- `zIndex: 25` — above the canvas (z:0) but below the boot sequence (z:50) and HUD (z:20 for buttons — but 25 is fine since the hint is non-interactive)
- bottom: 80px positions it above the HUD disc at bottom edge with 8px breathing room
- No new dependencies — pure React + sessionStorage
